<script src="https://cdn.jsdelivr.net/npm/@tensorflow-models/speech-commands@0.4.0/dist/speech-commands.min.js"></script>

<script>
const URL = "./my_model/";

// danh sách từ
const words = ["cat", "dog", "apple"];
let currentWordIndex = 0;

// đọc mẫu
function speak() {
    const word = words[currentWordIndex];
    const utterance = new SpeechSynthesisUtterance(word);
    utterance.lang = "en-US";
    speechSynthesis.speak(utterance);
}

// chuyển từ
function nextWord() {
    currentWordIndex = (currentWordIndex + 1) % words.length;
    document.getElementById("word").innerText = words[currentWordIndex];
    document.getElementById("result").innerText = "";
}

// load model
async function createModel() {
    const checkpointURL = URL + "model.json";
    const metadataURL = URL + "metadata.json";

    const recognizer = speechCommands.create(
        "BROWSER_FFT",
        undefined,
        checkpointURL,
        metadataURL
    );

    await recognizer.ensureModelLoaded();
    return recognizer;
}

// bắt đầu nghe
async function init() {
    const recognizer = await createModel();
    const classLabels = recognizer.wordLabels();

    recognizer.listen(result => {
        const scores = result.scores;
        const currentWord = words[currentWordIndex];

        let maxScore = 0;
        let predictedWord = "";

        // tìm từ AI đoán
        for (let i = 0; i < classLabels.length; i++) {
            if (scores[i] > maxScore) {
                maxScore = scores[i];
                predictedWord = classLabels[i];
            }
        }

        // chấm điểm
        let resultText = "";

        if (predictedWord === currentWord) {
            if (maxScore > 0.85) {
                resultText = "✅ Đúng (" + maxScore.toFixed(2) + ")";
            } else {
                resultText = "⚠️ Gần đúng (" + maxScore.toFixed(2) + ")";
            }
        } else {
            resultText = "❌ Sai (AI nghe: " + predictedWord + ")";
        }

        document.getElementById("result").innerText = resultText;

    }, {
        probabilityThreshold: 0.7,
        overlapFactor: 0.5
    });
}
</script>
